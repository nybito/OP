# Условие задачи:
Генератор простых чисел. Просуммируйте возвращаемые числа.

# Решение:
```Python
from functools import reduce  # Импортируем функцию reduce из библиотеки functools для последующей суммирования

def is_prime(n):
    """Проверка, является ли число простым."""
    if n <= 1:  # Если число меньше или равно 1, оно не является простым
        return False  # Возвращаем False
    for i in range(2, int(n**0.5) + 1):  # Проверяем делимость числа от 2 до квадратного корня из n
        if n % i == 0:  # Если n делится на i без остатка
            return False  # Значит n не простое, возвращаем False
    return True  # Если не нашли делителей, возвращаем True (число простое)

def generate_primes(limit):
    """Генерация простых чисел до заданного предела."""
    for num in range(2, limit):  # Проходим по числам от 2 до предела
        if is_prime(num):  # Проверяем, является ли текущее число простым
            yield num  # Если да, возвращаем генератор для простых чисел

def sum_primes(limit):
    """Суммирование простых чисел до заданного предела."""
    primes = generate_primes(limit)  # Генерируем простые числа до заданного предела
    return reduce(lambda x, y: x + y, list(primes))  # Подсчитываем сумму простых чисел с помощью функции reduce

# Пример использования
limit = 100  # Устанавливаем предел для генерации простых чисел
result = sum_primes(limit)  # Вызываем функцию, чтобы получить сумму простых чисел до предела
print(f"Сумма простых чисел до {limit}: {result}")  # Выводим результат на экран
```
