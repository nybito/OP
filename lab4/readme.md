Замыкание для получения очередной строки из файла. Декоратор, который будет логировать вызовы функций.
```Python
def log_calls(log_message="Функция вызвана"):
    """Декоратор для логирования вызовов функций с опциональным параметром."""
    
    def decorator(func):
        """Вложенная функция-декоратор, которая оборачивает целевую функцию."""
        
        def wrapper(*args, **kwargs):
            """Обертка для логирования вызовов функции."""
            # Выводим сообщение в консоль с информацией о вызове функции
            print(f"{log_message}: {func.__name__} с аргументами {args} и {kwargs}")
            # Вызываем оригинальную функцию и возвращаем её результат
            return func(*args, **kwargs)
        
        return wrapper  # Возвращаем обертку
    
    return decorator  # Возвращаем декоратор


def file_line_reader(file_path):
    """Замыкание для получения очередной строки из файла."""
    file = open(file_path, 'r')  # Открываем файл для чтения

    def get_next_line():
        """Функция для чтения следующей строки из файла."""
        line = file.readline()  # Читаем строку из файла
        if line:  # Если строка не пустая
            return line.strip()  # Убираем пробелы и возвращаем строку
        else:  # Если строка пустая
            return None  # Возвращаем None, если больше строк нет

    return get_next_line  # Возвращаем функцию для чтения строк


# Пример использования замыкания и декоратора
@log_calls("Чтение строки из файла")  # Применяем декоратор к функции
def read_line(get_next_line):
    """Функция для чтения строки, обернутая в декоратор."""
    return get_next_line()  # Вызываем замыкание для получения следующей строки


# Пример использования
if __name__ == "__main__":  # Проверяем, что скрипт запущен напрямую
    # Создаем замыкание для чтения строк из файла
    get_next_line = file_line_reader('example.txt')  # Укажите путь к вашему файлу

    # Читаем строки из файла
    for i in range(5):  # Прочитаем 5 строк, если они есть
        line = read_line(get_next_line)  # Вызываем функцию для чтения строки
        if line is not None:  # Если строка не пустая
            print(line)  # Выводим строку на экран
        else:  # Если строка пустая
            print("Больше строк нет.")  # Сообщаем, что строки закончились
            break  # Выходим из цикла

    # Закрываем файл
    get_next_line.__closure__[0].cell_contents.close()  # Закрываем файл, используя замыкание 
```
